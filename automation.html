from playwright.sync_api import Page, expect

from playwright.sync_api import Page, expect
from conftest import TEXT_30, fill_reason, get_apply_count


class TestApplyFlow:
    def test_successful_apply_shows_success_message(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        msg = page.locator("#formMessage")
        expect(msg).not_to_have_class("hidden")
        expect(msg).to_contain_text("정상적으로 접수")

    def test_successful_apply_increments_count(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        assert get_apply_count(page) == 1

    def test_successful_apply_disables_button(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        expect(page.locator("#submitBtn")).to_be_disabled()

    def test_successful_apply_changes_button_text(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        expect(page.locator("#submitBtn")).to_have_text("신청 완료")

    def test_duplicate_apply_shows_error(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        # disabled 버튼을 JS로 강제 재클릭하여 내부 상태 검증
        page.evaluate("""
            const btn = document.getElementById('submitBtn');
            btn.disabled = false;
            btn.click();
        """)
        expect(page.locator("#formMessage")).to_contain_text("이미 신청이 완료된 상태")

from playwright.sync_api import Page, expect
from conftest import TEXT_30, fill_reason, get_apply_count, get_char_count


class TestTabAndCancelFlow:
    def test_cancel_tab_shows_info_without_apply(self, page: Page):
        page.locator("#tabCancel").click()
        msg = page.locator("#cancelMessage")
        expect(msg).not_to_have_class("hidden")
        expect(msg).to_contain_text("신청된 내역이 없습니다")

    def test_cancel_tab_cancel_button_disabled_without_apply(self, page: Page):
        page.locator("#tabCancel").click()
        expect(page.locator("#cancelApplyBtn")).to_be_disabled()

    def test_cancel_tab_shows_submitted_reason(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        page.locator("#tabCancel").click()
        expect(page.locator("#submittedReason")).to_have_text(TEXT_30)

    def test_cancel_apply_decrements_count(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        assert get_apply_count(page) == 1
        page.locator("#tabCancel").click()
        page.locator("#cancelApplyBtn").click()
        assert get_apply_count(page) == 0

    def test_cancel_apply_shows_success_message(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        page.locator("#tabCancel").click()
        page.locator("#cancelApplyBtn").click()
        expect(page.locator("#cancelMessage")).to_contain_text("신청이 취소되었습니다")

    def test_cancel_resets_form(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        page.locator("#tabCancel").click()
        page.locator("#cancelApplyBtn").click()
        page.locator("#tabApply").click()
        expect(page.locator("#reason")).to_have_value("")
        expect(page.locator("#submitBtn")).to_be_disabled()
        assert get_char_count(page) == 0

    def test_reapply_after_cancel(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        page.locator("#tabCancel").click()
        page.locator("#cancelApplyBtn").click()
        page.locator("#tabApply").click()
        fill_reason(page, TEXT_30)
        expect(page.locator("#submitBtn")).to_be_enabled()
        page.locator("#submitBtn").click()
        assert get_apply_count(page) == 1

from playwright.sync_api import Page, expect
from conftest import TEXT_30, fill_reason, get_apply_count, get_char_count


class TestRegression:
    def test_coupon_state_independent_from_apply(self, page: Page):
        page.locator("#couponBtn").click()
        fill_reason(page, TEXT_30)
        expect(page.locator("#submitBtn")).to_be_enabled()

    def test_apply_state_independent_from_coupon(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        page.locator("#couponBtn").click()
        expect(page.locator("#couponMessage")).to_contain_text("쿠폰이 발급되었습니다")

    def test_tab_switch_preserves_applied_state(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        page.locator("#tabCancel").click()
        page.locator("#tabApply").click()
        expect(page.locator("#submitBtn")).to_be_disabled()
        expect(page.locator("#submitBtn")).to_have_text("신청 완료")

    def test_counter_not_negative_after_forced_double_cancel(self, page: Page):
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        page.locator("#tabCancel").click()
        page.locator("#cancelApplyBtn").click()
        # disabled 버튼 강제 재클릭
        page.evaluate("""
            const btn = document.getElementById('cancelApplyBtn');
            btn.disabled = false;
            btn.click();
        """)
        assert get_apply_count(page) >= 0

    def test_apply_tab_panel_visible_after_cancel_tab(self, page: Page):
        page.locator("#tabCancel").click()
        page.locator("#tabApply").click()
        expect(page.locator("#applyPanel")).not_to_have_class("hidden")
        expect(page.locator("#cancelPanel")).to_have_class("hidden")

    def test_full_flow_apply_cancel_reapply(self, page: Page):
        # 1차 신청
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        assert get_apply_count(page) == 1

        # 취소
        page.locator("#tabCancel").click()
        page.locator("#cancelApplyBtn").click()
        assert get_apply_count(page) == 0

        # 재신청
        page.locator("#tabApply").click()
        fill_reason(page, TEXT_30)
        page.locator("#submitBtn").click()
        assert get_apply_count(page) == 1
        expect(page.locator("#formMessage")).to_contain_text("정상적으로 접수")
